DEBUT DU PROG

	DEBUT DECLARATION VAR
		VAR TexteSource : T <- "";
		VAR TexteSelection : T <- "";
		VAR PositionDébut : N <- 0;
		VAR PositionFin : N <- 0;
		VAR TexteDébut : T <- "";
		VAR TexteFin : T <- "";
		VAR LongueurSource : N <- 0;
		VAR LongueurRecherche : N <- 0;
		VAR LongueurRemplacement : N <- 0;
		VAR Compteur : N <- 0;		
	FIN DECLARATION VAR

	Afficher : "Veuillez saisir un texte source."
	Saisir TexteSource
	Afficher : "Veuillez saisir la position de début."
	Saisir PositionDébut
	Afficher : "Veuillez saisir la position de fin."
	Saisir PositionFin

	LonguerSource = longueur(TexteSource)
	SI ((PositionDébut >= 1 ET PositionDébut <= LongueurSource) ET (PositionFin >= 2 ET PositionFin <= (LongueurSource + 1)))
		ALORS
			TANT QUE (Compteur <= (PositionFin - PositionDébut))

				TexteSelection = TexteSelection + caract(TexteSource, (PositionDébut + Compteur)

				Compteur++
			FIN TANT QUE

			Afficher : TexteSelection

		SINON
			Afficher : "Les positions ne sont pas possibles."
