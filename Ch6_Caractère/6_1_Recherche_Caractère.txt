6_1_Recherche_Caractère

DEBUT DU PROG

	DEBUT DECLARATION VAR
		VAR Texte : T <- "";
		VAR Caractère : T <- "";
		VAR LongueurTexte : N <- 0;
		VAR PremièrePosition : N <- 0;
		VAR DernièrePosition : N <- 0;
		VAR Compteur : N <- 1;
		VAR CompteurRépétition : N <- 0;		
	FIN DECLARATION VAR

	Afficher : "Veuillez saisir un mot."
	Saisir Texte
	Afficher : "Veuillez saisir un caractère."
	Saisir Caractère
	LongueurTexte = longueur(Texte)

	TANT QUE (Compteur <= LongueurTexte)

		SI (Caractère == caract(Texte, Compteur))
			ALORS
				SI (PremièrePosition == 0)
					ALORS
						PremièrePosition = Compteur
				FIN DE SI
				DernièrePosition = Compteur
				CompteurRépétition++
		FIN DE SI
		Compteur ++

	FIN TANT QUE

	SI (CompteurRépétion != 0)
		ALORS
			Afficher : "Votre caractère enregistré apparaît pour la première fois à  la position ", PremièrePosition, " et pour la dernière fois à la position ", DernièrePosition, ". Le caractère apparaît ", CompteurRépétition, " fois dans votre texte."
		SINON
			Afficher : "Le caractère enregistré n'est pas utilisé dans votre texte."
	FIN DE SI

FIN DU PROG
