DEBUT DU PROG

	DEBUT DECLARATION VAR
		VAR CompteurOccurenceMax : N <- 0;
		VAR CompteurOccurenceMin : N <- 0;
		VAR CompteurSaisie : N <- 0;
		VAR Saisie : N <- 0;
		VAR PlusPetit : N <- 0;
		VAR PlusGrand : N <- 0;
		VAR PremièrePositionPlusPetit : N <- 0;
		VAR PremièrePositionPlusGrand : N <- 0;
		VAR DernièrePositionPlusPetit : N <- 0;
		VAR DernièrePositionPlusGrand : N <- 0;
	FIN DECLARATION VAR

	Afficher : "Veuillez saisir des  valeurs, nous allons en faire une moyenne. Le programme prend fin avec un 100."
	Saisir Saisie
	SI (Saisie == 100)
		ALORS 	
			Afficher : "Pas de données à traiter. Bonne journée !"
	SINON		
		PlusPetit <- Saisie
		PlusGrand <- Saisie
		PremièrePositionPlusPetit <- 1
		DernièrePositionPlusPetit <- 1	
		PremièrePositionPlusGrand <- 1
		DernièrePositionPlusGrand <- 1
		CompteurOccurenceMax <- 1
		CompteurOccurenceMin <- 1
		Compteursaisie <- 1
		Saisir Saisie
		
		TANT QUE (Saisie != 100)	
			SI (Saisie <= PlusPetit)
				ALORS
					SI (Saisie == PlusPetit)
						ALORS
							CompteurOccurenceMin++
								FIN DE SI
					SINON
						CompteurOccurenceMin <- 1
						PremièrePositionPlusPetit <- CompteurSaisie + 1
						PlusPetit <- Saisie
					FIN DE SI
					DernièrePositionPlusPetit <- CompteurSaisie + 1
			SINON
				SI (Saisie == PlusGrand)
					ALORS
						CompteurOccurenceMax++
				SINON
					CompteurOccurenceMax <- 1
					PremièrePositionPlusGrand <- CompteurSaisie + 1
					PlusGrand <- Saisie
				FIN DE SI
				DernièrePositionPlusGrand <- CompteurSaisie + 1
		FIN DE SI
		CompteurSaisie++
		Saisir Saisie
	FIN TANT QUE

	Afficher : "Le plus petit nombre est ", PlusPetit, " et a été encodé ", CompteurOccurenceMin, " fois. La position du premier des plus petits nombres est ", PremièrePositionPlusPetit, " et celle du dernier est ", DernièrePositionPlusPetit.
	Afficher : "Le plus grand nombre est ", PlusGrand, " et a été encodé ", CompteurOccurenceMax, " fois. La position du premier des plus petits nombres est ", PremièrePositionPlusGrand, " et celle du dernier est ", DernièrePositionPlusGrand.
	Afficher : "Merci. Bonne journée!"

	FIN DE SI

FIN DU PROG
