4_2_Moyenne

DEBUT DU PROG

	DEBUT DECLARATION VAR
		VAR Saisie : N <- 0;
		VAR Somme : N <- 0;
		VAR Compteur : N <-0;
	FIN DECLARATION VAR

	Afficher : " Veuillez saisir des  valeurs, nous allons en faire une moyenne. Le programme prend fin avec un 100."

	TANT QUE (Saisie != 100)
		Saisir Saisie
		Somme = Somme + Saisie
		Compteur++
	FIN TANT QUE
	Afficher : "La moyenne des valeurs introduites est ", (Somme/Compteur)

FIN DU PROG
