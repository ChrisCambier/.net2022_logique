3_5_TVA

DEBUT DU PROG

	DEBUT DECLARATION VAR
		VAR Choix : N <- 0;
		VAR TVA : N <- 0;
		CONS Exit : N <- 0;

		VAR Somme : N <- 0;
	FIN DECLARATION VAR

	Afficher : "Veuillez sélectionner une TVA :
	(1) 6%
	(2) 12%
	(3) 21%
	(4) 0%
	(0) Sortie"

	Saisie TVA
	Afficher : "Veuillez saisir la somme à calculer"
	Saisie Somme

	CAS OU (Choix)
		CAS (1)
			TVA <- 6
		CAS (2)
			TVA <- 12
		CAS (3)
			TVA <- 21
		CAS (4)
			TVA <- 0
		CAS (0)
			Afficher : Une bonne journée !
		AUTRE CAS
			Afficher : Veuillez saisir une des options disponibles ! // Si on avait appris les boucles, on pourrait utiliser un Bool pour relancer la sélection
	FIN CAS OU

	Afficher : "Votre choix : ", Choix
	Afficher : "Prix (Hors TVA) : ", Somme
	Afficher : "Le taux de TVA vaut : ", TVA, "%"
	Afficher : "Le montant de la TVA est :", (Somme*TVA/100)
	Afficher : "Le montant de la TVAC est :", (Somme + (Somme*TVA/100))

FIN DU PROG
